// Funciones de Control Visual function showProg(val) { const cont = document.getElementById('progress-container'); const bar = document.getElementById('progress-bar'); cont.style.display = 'block'; bar.style.width = val + '%'; if(val >= 100) setTimeout(() => { cont.style.display = 'none'; bar.style.width = '0%'; }, 800); } function notify(msg) { const t = document.getElementById('toast'); t.innerText = msg; t.style.display = 'block'; setTimeout(() => { t.style.display = 'none'; }, 3000); } // 1. Guardar cambios con barra function save() { showProg(30); const newRates = { cr: document.getElementById('i-cr').value || rates.cr, ni: document.getElementById('i-ni').value || rates.ni, pa: document.getElementById('i-pa').value || rates.pa, hn: document.getElementById('i-hn').value || rates.hn, sv: document.getElementById('i-sv').value || rates.sv, gt: document.getElementById('i-gt').value || rates.gt, bz: document.getElementById('i-bz').value || rates.bz }; const newSets = { wa: document.getElementById('i-wa').value || sets.wa, tit: document.getElementById('i-tit').value || sets.tit, sinpe: document.getElementById('i-sinpe').value || sets.sinpe, iban: document.getElementById('i-iban').value || sets.iban }; showProg(60); db.ref('config').set({ rates: newRates, settings: newSets }).then(() => { showProg(100); notify("✅ CAMBIOS GUARDADOS EN LA NUBE"); }).catch(() => { showProg(0); notify("❌ ERROR AL GUARDAR"); }); } // 2. Subida de imágenes con progreso real function upImg(e) { const f = e.files[0]; if(!f) return; const ref = st.ref('laminas/' + Date.now()); const task = ref.put(f); task.on('state_changed', (snap) => { const pct = (snap.bytesTransferred / snap.totalBytes) * 100; showProg(pct); }, (err) => { notify("❌ ERROR DE CARGA"); }, () => { task.snapshot.ref.getDownloadURL().then(url => { db.ref('gallery').push({ url, fecha: hoy }); notify("✅ IMAGEN SUBIDA CON ÉXITO"); }); } ); } // 3. Alta y Baneo con confirmación function alta(id) { showProg(50); const d = new Date(); const clave = id.substring(3,7).toUpperCase() + String(d.getDate()).padStart(2,'0') + String(d.getMonth()+1).padStart(2,'0'); db.ref('users/'+id).update({ status: 'activo', key: clave }).then(() => { showProg(100); notify("✅ USUARIO ACTIVADO"); }); } function ban(id) { showProg(50); db.ref('users/'+id).update({ status: 'baneado' }).then(() => { showProg(100); notify("🚫 USUARIO BANEADO"); }); }
